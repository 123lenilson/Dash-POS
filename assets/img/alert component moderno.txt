    <script>
        let alertCounter = 0;

        function showAlert(type, message, duration = 5000) {
            const alertContainer = document.getElementById('alertContainer');
            const alertId = `alert-${++alertCounter}`;
            
            // Alert colors and icons
            const alertStyles = {
                success: {
                    bg: 'bg-green-50 border-green-200',
                    text: 'text-green-800',
                    icon: 'text-green-400',
                    iconPath: 'M5 13l4 4L19 7'
                },
                error: {
                    bg: 'bg-red-50 border-red-200',
                    text: 'text-red-800',
                    icon: 'text-red-400',
                    iconPath: 'M6 18L18 6M6 6l12 12'
                },
                warning: {
                    bg: 'bg-yellow-50 border-yellow-200',
                    text: 'text-yellow-800',
                    icon: 'text-yellow-400',
                    iconPath: 'M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-2.5L13.732 4c-.77-.833-1.964-.833-2.732 0L3.732 16.5c-.77.833.192 2.5 1.732 2.5z'
                },
                info: {
                    bg: 'bg-blue-50 border-blue-200',
                    text: 'text-blue-800',
                    icon: 'text-blue-400',
                    iconPath: 'M13 16h-1v-4h-1m1-4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z'
                }
            };

            const style = alertStyles[type] || alertStyles.info;
            
            const alertElement = document.createElement('div');
            alertElement.id = alertId;
            alertElement.className = `${style.bg} border-l-4 p-4 rounded-lg shadow-lg alert-enter`;
            
            alertElement.innerHTML = `
                <div class="flex items-start">
                    <div class="flex-shrink-0">
                        <svg class="w-5 h-5 ${style.icon}" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="${style.iconPath}"></path>
                        </svg>
                    </div>
                    <div class="ml-3 flex-1">
                        <p class="${style.text} text-sm font-medium">${message}</p>
                    </div>
                    <div class="ml-4 flex-shrink-0">
                        <button onclick="closeAlert('${alertId}')" class="${style.text} hover:opacity-75 transition-opacity">
                            <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path>
                            </svg>
                        </button>
                    </div>
                </div>
            `;
            
            alertContainer.appendChild(alertElement);
            
            // Auto-close after duration
            setTimeout(() => {
                closeAlert(alertId);
            }, duration);
        }

        function closeAlert(alertId) {
            const alertElement = document.getElementById(alertId);
            if (alertElement) {
                alertElement.classList.remove('alert-enter');
                alertElement.classList.add('alert-exit');
                
                setTimeout(() => {
                    if (alertElement.parentNode) {
                        alertElement.parentNode.removeChild(alertElement);
                    }
                }, 300);
            }
        }

        function createCustomAlert(event) {
            event.preventDefault();
            
            const type = document.getElementById('alertType').value;
            const message = document.getElementById('alertMessage').value.trim();
            
            if (!message) {
                showAlert('warning', 'Por favor, digite uma mensagem para o alert!');
                return;
            }
            
            showAlert(type, message);
            
            // Clear form
            document.getElementById('alertMessage').value = '';
        }

        // Show welcome message
        setTimeout(() => {
            showAlert('info', 'Bem-vindo! Teste os diferentes tipos de alertas.');
        }, 500);
    </script>
